        name: test_firewallrule config
        description: test firewall description
        action: deny
        priority: 0
        source_iptype: any
        destination_iptype: any
        direction: incoming
        protocol: tcp
        tcpflags:
          - syn

      trendmicro.deepsec.deepsec_firewallrules:
        state: absent
        name: test_firewallrule config
